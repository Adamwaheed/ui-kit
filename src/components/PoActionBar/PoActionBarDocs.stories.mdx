    import { useRouter } from 'vue-router';
    const router = useRouter();
    let actionBarItems = [
        {
            label: "New Staff",
            icon: PlusIcon,
        },
        {
            label: "Experiment",
            icon: BeakerIcon,
        },
    ];
    function handleActionBarClick(item) {
        switch (item) {
            case "New Staff":
                router.push('/staff/create');
                break;
            case "Experiment":
                console.log("exp");
                break;
            case "back":
                router.go(-1);
                break;
        }
    }
  `}
