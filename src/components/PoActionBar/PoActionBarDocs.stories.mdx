### With highlights
You can pass an optional `highlighted: true` to `items` to highlight the button. This is useful in cases where important action buttons are added to action bar.

  <Story id="layout-action-bar--with-highlight" />

<Source
  language='js'
  dark
  format={true}
  code={dedent`
    import { useRouter } from 'vue-router';
    const router = useRouter();
    let actionBarItems = [
        {
            label: "New Staff",
            icon: PlusIcon,
            highlighted: true
        },
        {
            label: "Experiment",
            icon: BeakerIcon,
        },
    ];
    function handleActionBarClick(item) {
        switch (item) {
            case "New Staff":
                router.push('/staff/create');
                break;
            case "Experiment":
                console.log("exp");
                break;
            case "back":
                router.go(-1);
                break;
        }
    }
  `}
/>
